Module Unit.Syntax

include module type of struct include Decl.Syntax end
type span = Decl.Syntax.span = {
start : Stdlib.Lexing.position;
stop : Stdlib.Lexing.position;
}
type ident = Yuujinchou.Trie.path option
type 'a node = 'a Decl.Syntax.node = {
node : 'a;
info : span option;
}
type t = t_ node
and t_ = Decl.Syntax.t_ =
| Ann of {
tm : t;
tp : t;
}
| Var of Yuujinchou.Trie.path
| Pi of t * ident * t
| Lam of ident * t
| App of t * t
| Sigma of t * ident * t
| Pair of t * t
| Fst of t
| Snd of t
| Univ of t
| VirPi of t * ident * t
| TpULvl
| ULvlTop
| ULvlShifted of t * int list
type cmd = cmd_ node
and cmd_ =
| Axiom of {
name : ident;
tp : t;
}
| Def of {
name : ident;
tm : t;
tp : t;
}
| Section of {
prefix : Yuujinchou.Trie.path;
block : section;
}
| Quit
and section = section_ node
and section_ = cmd list
type prog = section
